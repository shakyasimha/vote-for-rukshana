## 𑀓𑀸 𑀭𑀼𑀓𑁆𑀰𑀦𑀸 𑀓𑀧𑀸𑀮𑀻?

𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀻-𑀚𑀦𑀚𑀸𑀢𑀺-𑀅𑀥𑀺𑀓𑀸𑀭, 𑀦𑁂𑀯𑀸𑀭-𑀅𑀢𑁆𑀣, 𑀯𑀸𑀲𑀸𑀥𑀺𑀓𑀸𑀭, 𑀪𑀸𑀲𑀸𑀜𑀸𑀬, 𑀇𑀢𑁆𑀣𑀺𑀧𑀼𑀭𑀺𑀲-𑀯𑁂𑀫𑀢𑁆𑀢-𑀅𑀧𑁆𑀧𑀲𑀗𑁆𑀔𑁆𑀬-𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀅𑀝𑁆𑀞𑀸𑀬, 𑀲𑀸𑀥𑀸𑀭𑀺𑀡-𑀜𑀸𑀬𑀲𑁆𑀲 𑀅𑀪𑀺𑀦𑀻𑀳𑀸𑀭𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑀸, 𑀏𑀓𑁄 𑀫𑀸𑀦𑀼𑀲𑀺𑀓𑀸𑀥𑀺𑀓𑀸𑀭-𑀓𑀫𑁆𑀫𑀺𑀓𑁄 𑀳𑁄𑀢𑀺𑁇
𑀅𑀢𑁆𑀣-𑀪𑀜𑁆𑀚𑀦, 𑀆𑀡𑀢𑁆𑀢𑀺-𑀧𑀯𑁂𑀲𑀦𑁂𑀦 𑀘, 𑀯𑀺𑀲𑀫-𑀲𑀁𑀯𑀺𑀥𑀺𑀓𑀸𑀦𑀁 𑀲𑀗𑁆𑀖𑀝𑁆𑀝𑀦-𑀯𑀲𑁂𑀦, 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀦𑀲𑁆𑀲 𑀅𑀝𑁆𑀞𑀸𑀬 𑀩𑀳𑀼𑀯𑀲𑁆𑀲𑀸𑀦𑀺 𑀓𑀫𑁆𑀫𑀲𑀻𑀮𑁄, 𑀏𑀓𑁄 𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀓-𑀯𑀸𑀳𑀻 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑁆𑀣 𑀅𑀥𑀺𑀯𑀸𑀲𑀻-𑀞𑀸𑀦𑀻𑀬-𑀪𑀸𑀲𑀸𑀦𑀁 𑀭𑀚𑁆𑀚-𑀯𑀺𑀥𑀺𑀓-𑀪𑀸𑀲𑀸-𑀯𑀲𑁂𑀦 𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦𑀁, 𑀚𑀼𑀕𑀻-𑀲𑀫𑀼𑀤𑀸𑀬𑀲𑁆𑀲 𑀆𑀯𑀸𑀲-𑀅𑀥𑀺𑀓𑀸𑀭-𑀲𑀼𑀧𑀭𑀺𑀓𑁆𑀔𑀡𑀁, 𑀧𑀸𑀭𑀮𑁃𑀗𑁆𑀕𑀺𑀓-𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀲𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀺𑀥𑀺𑀓-𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦𑀁, 𑀲𑀺𑀓𑁆𑀔𑀸, 𑀓𑀫𑁆𑀫𑀦𑁆𑀢𑀸𑀦𑀁 𑀅𑀥𑀺𑀓𑀸𑀭-𑀲𑀺𑀚𑁆𑀛𑀦-𑀓𑀫𑁆𑀫𑀸𑀦𑀺 𑀲𑀦𑁆𑀢𑀺𑁇
𑀢𑀺𑀪𑀼𑀯𑀦-𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀬𑁂𑀦, 𑀪𑀸𑀲𑀸𑀯𑀺𑀚𑁆𑀚𑀸-𑀲𑀫𑀸𑀚𑀲𑀢𑁆𑀣𑁂𑀱𑀼 𑀲𑁆𑀦𑀸𑀢𑀓 (BA.) 𑀲𑀺𑀓𑁆𑀔𑀺𑀢𑁄𑁇 𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑀮-𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀬𑁂𑀦, 𑀧𑀜𑁆𑀘𑀯𑀲𑁆𑀲𑀺𑀓𑀸-𑀦𑀻𑀢𑀺-𑀲𑀸𑀔𑀸𑀲𑀼 𑀲𑁆𑀦𑀸𑀢𑀓 (BA.LLB) 𑀲𑀺𑀓𑁆𑀔𑀸 𑀧𑀭𑀺𑀦𑀺𑀝𑁆𑀞𑀺𑀢𑀸𑁇
𑀦𑁂𑀧𑀸𑀮𑀪𑀸𑀲𑀸, 𑀦𑁂𑀧𑀸𑀮𑀻, 𑀆𑀗𑁆𑀕𑀮𑁂𑀬, 𑀳𑀺𑀦𑁆𑀤𑀻, 𑀉𑀭𑁆𑀤𑀽, 𑀢𑀺𑀩𑁆𑀩𑀢-𑀪𑀸𑀲𑀸𑀲𑀼 𑀘, 𑀯𑀺𑀯𑀺𑀳-𑀪𑀸𑀲𑀸𑀦𑀁 𑀯𑀺𑀜𑁆𑀜𑀽, 𑀏𑀓𑁄 𑀪𑀸𑀲𑀸-𑀧𑀺𑀓𑁄𑁇 𑀣𑀸𑀬-𑀪𑀸𑀲𑀸𑀬 𑀆𑀭𑀫𑁆𑀪𑀺𑀓-𑀜𑀸𑀡-𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄𑁇 𑀦𑁂𑀧𑀸𑀮𑀲𑁆𑀲 𑀞𑀸𑀦𑀻𑀬-𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀻-𑀪𑀸𑀲𑀸𑀦𑀁 𑀲𑀺𑀓𑁆𑀔𑀦𑁂, 𑀩𑀼𑀚𑁆𑀛𑀦𑁂 𑀘, 𑀕𑀫𑁆𑀪𑀻𑀭-𑀅𑀥𑀺𑀫𑀼𑀢𑁆𑀢𑁄 𑀳𑁄𑀢𑀺𑁇

## 𑀭𑀼𑀓𑁆𑀰𑀡𑀸 𑀓𑀧𑀸𑀮𑀻𑀬𑀸 𑀧𑀝𑀺𑀪𑀽𑀫𑀺𑀓𑀸 𑀓𑀸?

𑀅𑀢𑁆𑀢𑀦𑁄 𑀦𑀺𑀭𑀼𑀤𑁆𑀥-𑀲𑀤𑁆𑀤𑀁 𑀫𑀼𑀔𑀭𑀺𑀢𑀼𑀁 𑀧𑀡𑁆𑀡𑀭𑀲-𑀯𑀲𑁆𑀲-𑀯𑀬𑁂𑀬𑁂𑀯 '𑀩𑁆𑀮𑀕'-𑀮𑁂𑀔𑀦-𑀫𑀼𑀔𑁂𑀦 𑀲𑀸𑀥𑀸𑀭𑀺𑀡-𑀅𑀪𑀺𑀬𑀸𑀦𑀲𑁆𑀲 𑀧𑀯𑁂𑀲𑀁 𑀓𑀢𑁆𑀯𑀸, 𑀅𑀦𑀼𑀓𑁆𑀓𑀫𑁂𑀦 𑀦𑁂𑀧𑀸𑀮𑀪𑀸𑀲𑀁 𑀯𑀺𑀚𑁆𑀚𑀸-𑀧𑀸𑀞𑀓𑀫𑁆𑀫𑁂 𑀧𑀭𑀺𑀬𑀸𑀧𑀼𑀡𑀦𑀁, 𑀦𑁆𑀳𑀽𑀧𑀼𑀔𑀽-𑀲𑀭𑀁 (𑀭𑀸𑀦𑀻𑀧𑁄𑀔𑀭𑀻) 𑀅𑀢𑁆𑀢𑀦𑁄 𑀫𑀽𑀮𑁂 𑀯𑀺𑀥𑀸𑀦𑁂𑀦 𑀧𑀼𑀦-𑀦𑀺𑀫𑀸𑀡𑀦𑀁, 𑀫𑀕𑁆𑀕-𑀯𑀺𑀢𑁆𑀣𑀸𑀭-𑀓𑀸𑀭𑀡𑀸 𑀦𑁂𑀯𑀸𑀭-𑀲𑀫𑀼𑀤𑀸𑀬𑀲𑁆𑀲 𑀉𑀘𑁆𑀙𑁂𑀤𑀦-𑀦𑀺𑀭𑁄𑀥𑀦𑀁, 𑀘 𑀪𑀸𑀲𑀸𑀜𑀸𑀬𑀲𑁆𑀲 𑀲𑀼𑀦𑀺𑀘𑁆𑀙𑀬-𑀯𑀺𑀳𑀸𑀦𑁂𑀲𑀼 𑀲𑀓𑁆𑀭𑀺𑀬-𑀅𑀢𑁆𑀣-𑀪𑀜𑁆𑀚𑀦𑀁 𑀦𑀸𑀬𑀓-𑀪𑀽𑀫𑀺𑀓𑀸 𑀘 𑀦𑀺𑀩𑁆𑀩𑀢𑀺𑀢𑀯𑀸𑀦𑁆𑁇
𑀅𑀦𑀼𑀯𑀲𑁆𑀲𑀺𑀓-𑀕𑁅𑀭𑀯-𑀬𑀸𑀢𑁆𑀭𑀸 (𑀧𑁆𑀭𑀸𑀇𑀟 𑀧𑀭𑁂𑀟) 𑀯𑀺𑀥𑀸𑀦𑁂𑀦, 𑀯𑀺𑀲𑁂𑀲𑁂𑀦 𑀧𑀸𑀭𑀮𑁃𑀗𑁆𑀕𑀺𑀓-𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀲𑀜𑁆𑀜𑀸𑀡-𑀅𑀥𑀺𑀓𑀸𑀭𑀲𑁆𑀲, 𑀇𑀢𑁆𑀣𑀺𑀧𑀼𑀭𑀺𑀲-𑀯𑁂𑀫𑀢𑁆𑀢-𑀅𑀧𑁆𑀧𑀲𑀗𑁆𑀔𑁆𑀬-𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀯𑀺𑀥𑀺𑀓-𑀲𑀳𑀸𑀬𑀸𑀬 𑀘 𑁨𑁦𑁭𑁫 𑀯𑀲𑁆𑀲𑁂 '𑀓𑁆𑀯𑀬𑁂𑀭 𑀬𑀼𑀣 𑀕𑁆𑀭𑀼𑀧' (Queer Youth Group) 𑀇𑀢𑀺 𑀆𑀬𑀢𑀦𑀲𑁆𑀲 𑀲𑀳-𑀲𑀫𑁆𑀧𑀦𑁆𑀦𑁄 𑀅𑀳𑁄𑀲𑀺𑁇 𑀩𑀮𑀸𑀓𑁆𑀓𑀸𑀭-𑀯𑀺𑀥𑀺𑀫𑁆𑀳𑀺 𑀲𑀩𑁆𑀩-𑀲𑀫𑀸𑀯𑁂𑀲𑀦𑀁, 𑀯𑀺𑀯𑀳-𑀲𑀫𑀸𑀦𑀢𑀸, 𑀦𑀸𑀕𑀭𑀻𑀓𑀢𑀸-𑀅𑀥𑀺𑀓𑀸𑀭𑀁, 𑀭𑀚𑁆𑀚-𑀲𑁂𑀯𑀸𑀲𑀼 𑀆𑀭𑀓𑁆𑀔𑀡𑀁, 𑀲𑀺𑀓𑁆𑀔𑀸-𑀧𑀯𑁂𑀲𑀦𑀁 𑀘, 𑀇𑀢𑀺 𑀯𑀺𑀯𑀺𑀳-𑀮𑁄𑀓𑀸𑀳𑀺𑀢-𑀅𑀢𑁆𑀣-𑀦𑀺𑀯𑁂𑀤𑀦𑁂𑀲𑀼 '𑀓𑁆𑀯𑀬𑁂𑀭 𑀬𑀼𑀣 𑀕𑁆𑀭𑀼𑀧' 𑀇𑀢𑀺 𑀲𑀗𑁆𑀖𑁂𑀦 𑀦𑀸𑀬𑀓𑀢𑁆𑀢𑀁 𑀕𑀳𑀺𑀢𑀁𑁇
𑁨𑁦𑁭𑁭 𑀯𑀲𑁆𑀲𑀸𑀢𑁄 𑀢𑀲𑁆𑀲 𑀭𑀣𑀺𑀬𑀸-𑀲𑀗𑁆𑀕𑀸𑀫𑁄 𑀦𑁆𑀬𑀸𑀬𑀸𑀮𑀬𑁂𑀲𑀼 𑀯𑀺𑀢𑁆𑀣𑀸𑀭𑀺𑀢𑁄𑁇 𑀏𑀢𑁂𑀦𑁂𑀯 𑀅𑀦𑀼𑀓𑁆𑀓𑀫𑁂𑀦 𑀧𑀼𑀩𑁆𑀩𑀜𑁆𑀘𑀮-𑀯𑀺𑀚𑁆𑀚𑀸𑀮𑀬𑁂 𑀲𑀺𑀓𑁆𑀔𑀓-𑀧𑀜𑁆𑀜𑀸𑀧𑀓𑀸𑀬𑀓𑀁 (𑀭𑀚𑀺𑀱𑁆𑀝𑁆𑀭𑁂𑀰𑀦) 𑀲𑀯𑁆𑀯𑀼𑀘𑁆𑀘-𑀦𑁆𑀬𑀸𑀬𑀸𑀮𑀬𑀸 𑀅𑀦𑁆𑀢𑀭𑀺𑀫-𑀆𑀡𑀢𑁆𑀢𑀺𑀁 𑀮𑀤𑁆𑀥𑀼𑀁 𑀲𑀫𑀢𑁆𑀣𑁄 𑀳𑀼𑀢𑁆𑀯𑀸, 𑀇𑀤𑀸𑀦𑀺 𑀇𑀢𑁆𑀣𑀺𑀧𑀼𑀭𑀺𑀲-𑀯𑁂𑀫𑀢𑁆𑀢-𑀅𑀧𑁆𑀧𑀲𑀗𑁆𑀔𑁆𑀬-𑀲𑀫𑀼𑀤𑀸𑀬𑀲𑁆𑀲 𑀘 𑀅𑀜𑁆𑀜𑁂𑀲𑀁𑀧𑀺 𑀲𑀻𑀫𑀸𑀧𑁂𑀢-𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀳𑀺𑀢𑀲𑀼𑀔𑀸𑀬, 𑀯𑀺𑀥𑀺𑀓-𑀜𑀸𑀬𑀲𑁆𑀲 𑀘 𑀅𑀢𑁆𑀣𑀸𑀬 𑀦𑀺𑀭𑀦𑁆𑀢𑀭𑀁 𑀓𑀫𑁆𑀫𑀲𑀻𑀮𑁄 𑀳𑁄𑀢𑀺𑁇
𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣-𑀦𑀺𑀯𑁂𑀤𑀦𑁂𑀲𑀼 𑀭𑀚𑁆𑀚-𑀓𑀺𑀘𑁆𑀘-𑀪𑀸𑀲𑀸-𑀯𑀲𑁂𑀦 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀻-𑀞𑀸𑀦𑀻𑀬-𑀪𑀸𑀲𑀸𑀦𑀁 𑀧𑀝𑁆𑀝𑀧𑀦𑀁 𑀧𑀬𑁄𑀕𑁄 𑀘 𑀲𑀦𑁆𑀢𑀺𑁇
𑁨𑁦𑁭𑁯 𑀯𑀲𑁆𑀲𑁂 𑀞𑀸𑀦𑀻𑀬-𑀧𑀲𑀸𑀲𑀦𑁂𑀦 𑀢𑀲𑁆𑀲 𑀓𑀼𑀮𑀁 𑀅𑀢𑁆𑀢𑀦𑁄 𑀧𑁄𑀭𑀸𑀡𑀓-𑀖𑀭𑀸 𑀯𑀺𑀦𑀸𑀲𑀺𑀢𑀁, 𑀢𑀢𑁄 𑀧𑀭𑀁 '𑀣𑀸𑀬𑁆𑀩𑀸𑀬𑁆𑀬𑀸 𑀮𑁆𑀯𑀸𑀧𑀼' 𑀇𑀢𑀺 𑀅𑀪𑀺𑀬𑀸𑀦𑀁 𑀆𑀭𑀩𑁆𑀥𑀁, 𑀚𑀼𑀕𑀻-𑀲𑀫𑀼𑀤𑀸𑀬𑀲𑁆𑀲 𑀧𑀸𑀭𑀫𑁆𑀧𑀭𑀺𑀓-𑀆𑀯𑀸𑀲𑀲𑁆𑀲 𑀲𑀸𑀫𑀺𑀓-𑀅𑀥𑀺𑀓𑀸𑀭𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀘 𑀯𑀘𑁆𑀘𑀢𑀺𑁇
𑀯𑀓𑁆𑀭𑀧𑀤 (𑀓𑁆𑀮𑀩𑀨𑀼𑀝) 𑀇𑀢𑀺 𑀓𑀸𑀬𑀺𑀓-𑀯𑀺𑀓𑀮𑀢𑀸𑀬 𑀲𑀳 𑀚𑀸𑀢𑁄, 𑀇𑀤𑀸𑀦𑀺 𑀅𑀯𑀲𑁂𑀲-𑀯𑀺𑀓𑀮𑀢𑀸𑀬 (𑀭𑁂𑀲𑀺𑀟𑁆𑀬𑀽𑀅𑀮) 𑀲𑀳 𑀚𑀻𑀯𑀫𑀸𑀦𑁄, 𑀲𑁄 𑀲𑀼𑀧𑁆𑀧𑀯𑁂𑀲𑀢𑀸𑀬 𑀯𑀺𑀓𑀮-𑀅𑀗𑁆𑀕-𑀧𑀼𑀕𑁆𑀕𑀮𑀸𑀦𑀁 𑀚𑀻𑀯𑀦𑁆𑀢-𑀅𑀦𑀼𑀪𑀯𑀁 𑀭𑀸𑀚𑀓𑀻𑀬-𑀫𑀜𑁆𑀘𑁂 𑀆𑀦𑁂𑀢𑀼𑀁 𑀇𑀘𑁆𑀙𑀢𑀺𑁇
𑀢𑁂𑀦 𑀇𑀢𑁆𑀣𑀺𑀧𑀼𑀭𑀺𑀲𑀢𑀸, 𑀓𑀸𑀫𑀢𑀸, 𑀦𑁂𑀧𑀸𑀮𑀪𑀸𑀲𑀸-𑀯𑀺𑀲𑀬𑁂𑀲𑀼 𑀘 𑀢𑀻𑀲𑀼 𑀪𑀸𑀲𑀸𑀲𑀼 𑀯𑀻𑀲𑀢𑀺-𑀅𑀥𑀺𑀓-𑀧𑁄𑀢𑁆𑀣𑀓𑀸𑀦𑀺 𑀮𑀺𑀔𑀺𑀢𑀸𑀦𑀺𑁇

## 𑀭𑀼𑀓𑁆𑀰𑀡𑀸𑀬𑀸 𑀫𑀼𑀔𑁆𑀬𑀸 𑀲𑀺𑀤𑁆𑀥𑀺𑀬𑁄 𑀓𑀸-𑀓𑀸?

-**𑀲𑀫𑁆𑀫𑀸𑀦𑀁 𑀘 𑀧𑀭𑀺𑀓𑁆𑀔𑀸𑀭𑁄 𑀘 :** 𑀫𑀸𑀦𑀼𑀲𑀺𑀓𑀸𑀥𑀺𑀓𑀸𑀭-𑀔𑁂𑀢𑁆𑀢𑁂 𑀓𑀢-𑀧𑀭𑀺𑀯𑀢𑁆𑀢𑀓-𑀬𑁄𑀕𑀤𑀸𑀦𑀲𑁆𑀲 𑀫𑀸𑀦𑀦𑀢𑁆𑀣𑀁, 𑀢𑁂𑀦 𑀩𑀺𑀩𑀺𑀲𑀻 𑁧𑁦𑁦 𑀫𑀳𑀺𑀮𑀸 (𑀲𑀦𑁆 𑁨𑁦𑁨𑁩), 𑀝𑀸𑀇𑀫 𑁧𑁦𑁦 𑀦𑁂𑀓𑁆𑀲𑁆𑀝 (𑀲𑀦𑁆 𑁨𑁦𑁨𑁪), 𑀘 𑀨𑁄𑀭𑁆𑀩𑁆𑀲 𑁩𑁦 𑀅𑀦𑁆𑀟𑀭 𑁩𑁦 (𑀲𑀦𑁆 𑁨𑁦𑁨𑁫), 𑀇𑀢𑀺 𑀢𑀻𑀡𑀺 𑀧𑀲𑀺𑀞𑁆𑀞𑀸𑀦𑀺 𑀅𑀦𑁆𑀢𑀭𑀭𑀚𑁆𑀚𑀺𑀬-𑀲𑀫𑁆𑀫𑀸𑀦𑀦𑀺 𑀮𑀤𑁆𑀥𑀸𑀦𑀺 𑀲𑀦𑁆𑀢𑀺𑁇

-**𑀯𑀺𑀥𑀺𑀓-𑀲𑀻𑀫𑀸𑀦𑁆𑀢-𑀧𑀸𑀲𑀸𑀡𑁄:** 𑀦𑁂𑀧𑀸𑀮𑀭𑀚𑁆𑀚𑁂 𑀧𑀸𑀭𑀮𑁃𑀗𑁆𑀕𑀺𑀓-𑀲𑀜𑁆𑀜𑀸𑀡𑀲𑁆𑀲 𑀯𑀺𑀥𑀺𑀓-𑀲𑀫𑁆𑀫𑀦𑁆𑀦𑀦-𑀇𑀢𑀺𑀳𑀸𑀲𑀫𑁆𑀳𑀺, 𑀢𑀲𑁆𑀲 𑀅𑀢𑁆𑀣-𑀦𑀺𑀯𑁂𑀤𑀦𑁂 𑀲𑀯𑁆𑀯𑀼𑀘𑁆𑀘-𑀦𑁆𑀬𑀸𑀬𑀸𑀮𑀬𑀸 𑁨𑁦𑁮𑁦 𑀯𑀲𑁆𑀲𑁂 𑀓𑀢𑁄 𑀯𑀺𑀦𑀺𑀘𑁆𑀙𑀬𑁄 𑀏𑀓𑁄 𑀯𑀺𑀥𑀺𑀓-𑀲𑀻𑀫𑀸𑀦𑁆𑀢-𑀧𑀸𑀲𑀸𑀡𑁄 𑀯𑀺𑀬 𑀧𑀝𑁆𑀝𑀧𑀺𑀢𑁄 𑀅𑀳𑁄𑀲𑀺𑁇

---

## 𑀭𑀼𑀓𑁆𑀰𑀡𑀸-𑀓𑀧𑀸𑀮𑀻-𑀫𑀳𑁄𑀤𑀬𑀸𑀬 𑀓𑀣𑀁 𑀫𑀢𑀤𑀸𑀦𑀁 𑀓𑀸𑀢𑀩𑁆𑀩𑀁?

𑀭𑀼𑀓𑁆𑀰𑀡𑀸 𑀓𑀧𑀸𑀮𑀻, 𑀧𑁆𑀭𑀕𑀢𑀺𑀲𑀻𑀮-𑀮𑁄𑀓𑀢𑀸𑀦𑁆𑀢𑀺𑀓-𑀧𑀓𑁆𑀔𑀢𑁄, 𑀧𑀥𑀺𑀦𑀺𑀥𑀺𑀲𑀪𑀸𑀬 𑀏𑀓𑀸 𑀫𑀼𑀔𑁆𑀬𑀸 𑀲𑀫𑀸𑀦𑀼𑀧𑀸𑀢𑀺𑀓-𑀅𑀪𑀺𑀢𑁆𑀣𑀺𑀢𑀸 𑀳𑁄𑀢𑀺𑁇 𑀏𑀢𑀲𑁆𑀲 𑀧𑀓𑁆𑀔𑀲𑁆𑀲 𑀦𑀺𑀭𑁆𑀯𑀸𑀘𑀦-𑀘𑀺𑀡𑁆𑀡𑀁 "𑀅𑀓𑁆𑀔𑀺" (𑀘𑀓𑁆𑀔𑀼) 𑀇𑀢𑀺 𑀅𑀢𑁆𑀣𑀺𑁇
𑀭𑀼𑀓𑁆𑀰𑀡𑀸-𑀫𑀳𑁄𑀤𑀬𑀸𑀬 𑀫𑀢𑀤𑀸𑀦𑀁 𑀓𑀸𑀢𑀼𑀁, 𑀲𑀫𑀸𑀦𑀼𑀧𑀸𑀢𑀺𑀓-𑀫𑀢𑁆𑀢𑀓𑁂, "𑀅𑀓𑁆𑀔𑀺" 𑀘𑀺𑀡𑁆𑀡𑀫𑁆𑀳𑀺 𑀲𑁄𑀢𑁆𑀣𑀺𑀓-𑀫𑀼𑀤𑁆𑀤𑀁 𑀤𑀢𑁆𑀯𑀸, 𑀫𑀢𑀤𑀸𑀦𑀁 𑀓𑀭𑁄𑀣𑁇
𑀲𑀫𑀸𑀦𑀼𑀧𑀸𑀢𑀺𑀓-𑀦𑀺𑀭𑁆𑀯𑀸𑀘𑀦-𑀧𑀭𑀺𑀬𑀸𑀬𑁂𑀦 𑀅𑀪𑀺𑀢𑁆𑀣𑀺𑀢𑀸 𑀅𑀢𑁆𑀣𑀺, 𑀢𑀲𑁆𑀫𑀸 𑀲𑀩𑁆𑀩-𑀤𑁂𑀲𑀲𑁆𑀲 𑀬𑁂𑀲𑀁 𑀓𑁂𑀲𑀁 𑀘 𑀦𑀺𑀭𑁆𑀯𑀸𑀘𑀦-𑀔𑁂𑀢𑁆𑀢𑀸𑀦𑀁 𑀫𑀢𑀤𑀸𑀢𑀸𑀭𑁄, 𑀅𑀢𑁆𑀢𑀦𑁄 𑀫𑀢𑁆𑀢𑀓𑁂 "𑀅𑀓𑁆𑀔𑀺" 𑀘𑀺𑀡𑁆𑀡𑀁 𑀫𑀢𑀁 𑀤𑀢𑁆𑀯𑀸, 𑀢𑀁 𑀉𑀧𑀢𑁆𑀣𑀫𑁆𑀪𑀦𑀁 𑀓𑀸𑀢𑀼𑀁 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺𑁇

---

## 𑀭𑀼𑀓𑁆𑀰𑀡𑀸-𑀫𑀳𑁄𑀤𑀬𑀸𑀬 𑀓𑀺𑀫𑀢𑁆𑀣𑀁 𑀫𑀢𑀁 𑀤𑀸𑀢𑀩𑁆𑀩𑀁?

“'𑀫𑀕𑁆𑀕𑀁 𑀓𑀭𑀺𑀲𑁆𑀲𑀸𑀫𑀻' 𑀢𑀺 𑀲𑀩𑁆𑀩𑁂𑀯 𑀯𑀤𑀦𑁆𑀢𑀺𑁇 𑀧𑀦 𑀢𑀲𑁆𑀲 𑀫𑀕𑁆𑀕𑀲𑁆𑀲 𑀦𑀺𑀫𑀸𑀡-𑀓𑀸𑀮𑁂, 𑀢𑀢𑁆𑀣 𑀞𑀸𑀦𑀺𑀓𑀸𑀦𑀁 𑀅𑀥𑀺𑀯𑀢𑁆𑀣𑀻-𑀚𑀦𑀢𑀸𑀦𑀁 𑀘 𑀕𑁆𑀭𑀸𑀲-𑀯𑀸𑀲𑁂 𑀓𑁄 𑀉𑀧𑀖𑀸𑀢𑁄 𑀳𑁄𑀢𑀺? 𑀓𑀺𑀁 𑀲𑁄 𑀫𑀕𑁆𑀕𑁄 𑀩𑀸𑀮𑀓𑀸𑀦𑀁 𑀅𑀢𑁆𑀣𑀸𑀬 𑀲𑀼𑀭𑀓𑁆𑀔𑀺𑀢𑁄 𑀅𑀢𑁆𑀣𑀺? 𑀓𑀺𑀁 𑀯𑀼𑀟𑁆𑀠-𑀚𑀦𑀸 𑀢𑀲𑁆𑀫𑀺 𑀫𑀕𑁆𑀕𑁂 𑀦𑀺𑀲𑁆𑀲𑀗𑁆𑀓𑀁 𑀯𑀺𑀘𑀭𑀺𑀢𑀼𑀁 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺? 𑀓𑀺𑀁 𑀯𑀺𑀓𑀮-𑀅𑀗𑁆𑀕-𑀧𑀼𑀕𑁆𑀕𑀮𑀸 𑀫𑀼𑀢𑁆𑀢-𑀯𑀲𑁂𑀦 𑀢𑀁 𑀫𑀕𑁆𑀕𑀁 𑀉𑀧𑀪𑀼𑀜𑁆𑀚𑀺𑀢𑀼𑀁 𑀲𑀓𑁆𑀓𑁄𑀦𑁆𑀢𑀺? 𑀫𑀬𑀸 𑀏𑀢𑁂𑀲𑀁 𑀬𑁂𑀯 𑀅𑀧𑀲𑁆𑀲𑀓-𑀦𑀺𑀭𑀼𑀤𑁆𑀥-𑀧𑀓𑁆𑀔𑀸𑀦𑀁 𑀲𑀪𑀸𑀯𑀺𑀬𑀁 (𑀲𑀤𑀦𑀫𑁆𑀳𑀺) 𑀉𑀝𑁆𑀞𑀸𑀧𑀦𑀢𑁆𑀣𑀁 𑀅𑀪𑀺𑀢𑁆𑀣𑀺𑀢𑀁𑁇 𑀲𑀸𑀥𑀸𑀭𑀺𑀡-𑀜𑀸𑀬𑀲𑁆𑀲 𑀲𑀼𑀦𑀺𑀘𑁆𑀙𑀬-𑀓𑀭𑀡𑀁, 𑀭𑀚𑁆𑀚𑀲𑁆𑀲 𑀫𑀽𑀮𑀥𑀸𑀭𑀸𑀬 𑀑𑀛𑀸𑀬𑀺𑀢-𑀲𑀻𑀫𑀸𑀦𑁆𑀢𑀓𑀾𑀢-𑀲𑀫𑀼𑀤𑀸𑀬𑀲𑁆𑀲 𑀧𑀥𑀺𑀦𑀺𑀥𑀺𑀢𑁆𑀢𑀁 𑀘 𑀫𑀬𑁆𑀳𑀁 𑀫𑀼𑀔𑁆𑀬𑁄 𑀅𑀢𑁆𑀣𑁄 𑀳𑁄𑀢𑀺𑁇”

**𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀲𑀜𑁆𑀜𑀸𑀡𑀁, 𑀲𑀩𑁆𑀩𑁂𑀲𑀁 𑀫𑀸𑀦𑀦𑀁 : 𑀲𑀸𑀥𑀸𑀭𑀺𑀡-𑀜𑀸𑀬𑀲𑁆𑀲 𑀅𑀢𑁆𑀣𑀸𑀬 𑀫𑀬𑁆𑀳𑀁 𑀅𑀪𑀺𑀬𑀸𑀦𑀁𑁇**
